2.获取日期
a.var year=MyData.getFullYear();获取年
b.var month=MyData.getMonth()+1;获取月
c.var da=MyData.getDate();获取日
d.var day=MyData.getDay();获取周
0~6，0代表周日
3.获取时间
a.var hour=MyData.getHours();获取小时
0~23
b.var min=MyData.getMinutes();获取分钟
0~59
c.var msec=MyData.getSeconds();获取秒
0~59
d.var ms=MyData.getMilliseconds();获取毫秒
0~999
4.获取完整日期、时间
a.var d=MyData.toLocaleDateString();获取完整日期
b.var t=MyData.toLocaleTimeString();获取完整时间
c.var dt=MyData.toLocaleString();获取完整日期、时间

var ts=MyData.getTime();获取时间戳
